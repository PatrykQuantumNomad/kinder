       ```go
       package dev

       import (
           "fmt"
           "os"

           "sigs.k8s.io/kind/pkg/cluster"
       )

       // WriteKubeconfigTemp obtains the cluster's kubeconfig (using the
       // host-reachable external endpoint) and writes it to a unique temp
       // file with mode 0600 (kubeconfigs contain client cert+key — they are
       // credentials per RESEARCH security V4).
       //
       // Returns the file path and a cleanup function the caller MUST defer
       // (registers os.Remove). On any error, no file is left behind.
       //
       // Per RESEARCH A4: provider.KubeConfig(name, false) returns the EXTERNAL
       // (host-reachable) kubeconfig — the boolean parameter is `internal`,
       // so false = NOT internal = external. Verified in pkg/cluster/provider.go:220.
       func WriteKubeconfigTemp(provider *cluster.Provider, clusterName string) (path string, cleanup func(), err error) {
           cfg, err := provider.KubeConfig(clusterName, false)
           if err != nil {
               return "", nil, fmt.Errorf("get kubeconfig for cluster %q: %w", clusterName, err)
           }
           f, err := os.CreateTemp("", "kinder-dev-*.kubeconfig")
           if err != nil {
               return "", nil, fmt.Errorf("create kubeconfig tempfile: %w", err)
           }
           // Tighten permissions BEFORE writing — even though CreateTemp creates
           // 0600 on most platforms, be explicit (V4 mitigation; defensive on
           // umask edge cases).
           if err := os.Chmod(f.Name(), 0600); err != nil {
               f.Close()
               os.Remove(f.Name())
               return "", nil, fmt.Errorf("chmod kubeconfig: %w", err)
           }
           if _, err := f.WriteString(cfg); err != nil {
               f.Close()
               os.Remove(f.Name())
               return "", nil, fmt.Errorf("write kubeconfig: %w", err)
           }
           if err := f.Close(); err != nil {
               os.Remove(f.Name())
               return "", nil, fmt.Errorf("close kubeconfig: %w", err)
           }
           cleanup = func() { os.Remove(f.Name()) }
           return f.Name(), cleanup, nil
       }
       ```

       ```go
       package dev

       import (
           "context"
           "fmt"
           "io"
           "os"
           "path/filepath"

           "sigs.k8s.io/kind/pkg/cluster"
           "sigs.k8s.io/kind/pkg/cluster/nodes"
           "sigs.k8s.io/kind/pkg/cluster/nodeutils"
           "sigs.k8s.io/kind/pkg/errors"
           "sigs.k8s.io/kind/pkg/exec"
           "sigs.k8s.io/kind/pkg/fs"
           "sigs.k8s.io/kind/pkg/log"
       )

       // ImageLoaderFn is the per-node image loader signature, matching the
       // signature of nodeutils.LoadImageArchiveWithFallback(node, openTar).
       // Tests inject a fake to capture calls without spinning a cluster.
       type ImageLoaderFn func(node nodes.Node, openTar func() (io.ReadCloser, error)) error

       // LoadOptions carries inputs to LoadImagesIntoCluster. Each field is
       // explicit so the cycle runner (Plan 03) can build a fresh options
       // struct per cycle without sharing state.
       type LoadOptions struct {
           ClusterName   string
           ImageTag      string                  // single image (kinder dev builds ONE per cycle)
           BinaryName    string                  // docker | podman | nerdctl
           Logger        log.Logger
           Provider      *cluster.Provider
           // ImageLoaderFn is the per-node loader. nil → use the production
           // nodeutils.LoadImageArchiveWithFallback. Tests pass a fake.
           ImageLoaderFn ImageLoaderFn
       }

       // LoadImagesIntoCluster imports the named image into every cluster
       // node, replicating the core pipeline of `kinder load images` against
       // public APIs (RESEARCH §1 Option A — extracts the logic without
       // calling the unexported runE in pkg/cmd/kind/load/images/images.go).
       //
       // Pipeline (matches images.go runE behavior):
       //   1. Inspect imageTag locally → image ID.
       //   2. provider.ListInternalNodes(clusterName) → all cluster nodes.
       //   3. For each node: nodeutils.ImageTags → if ID present but tag
       //      missing, ReTagImage; if absent, mark node as candidate.
       //   4. If no candidates → return nil (smart-skip path: image already
       //      everywhere with correct tag).
       //   5. fs.TempDir + exec save → write image TAR to host temp.
       //   6. errors.UntilErrorConcurrent: per-candidate-node, call
       //      ImageLoaderFn(node, openTar) → import.
       func LoadImagesIntoCluster(ctx context.Context, opts LoadOptions) error {
           if opts.ImageTag == ""    { return fmt.Errorf("dev load: ImageTag empty") }
           if opts.ClusterName == "" { return fmt.Errorf("dev load: ClusterName empty") }
           if opts.BinaryName == ""  { return fmt.Errorf("dev load: BinaryName empty") }

           loader := opts.ImageLoaderFn
           if loader == nil { loader = nodeutils.LoadImageArchiveWithFallback }

           // 1. Inspect image ID
           imageID, err := imageInspectID(ctx, opts.BinaryName, opts.ImageTag)
           if err != nil {
               return fmt.Errorf("inspect image %q: %w", opts.ImageTag, err)
           }

           // 2. List nodes
           internalNodes, err := opts.Provider.ListInternalNodes(opts.ClusterName)
           if err != nil { return fmt.Errorf("list nodes: %w", err) }

           // 3. Smart-skip + candidate selection
           var candidates []nodes.Node
           for _, n := range internalNodes {
               tags, err := nodeutils.ImageTags(n, imageID)
               if err != nil {
                   // Node not reachable / not running → treat as candidate so we attempt
                   candidates = append(candidates, n)
                   continue
               }
               if _, hasTag := tags[opts.ImageTag]; hasTag {
                   continue // already has correct tag
               }
               if len(tags) > 0 {
                   // Image present under a different tag → re-tag in place (faster than re-import)
                   if err := nodeutils.ReTagImage(n, imageID, opts.ImageTag); err != nil {
                       opts.Logger.Warn(fmt.Sprintf("retag on node %s failed, will reimport: %v", n.String(), err))
                       candidates = append(candidates, n)
                   }
                   continue
               }
               candidates = append(candidates, n)
           }
           if len(candidates) == 0 {
               return nil // smart-skip: image already on every node with correct tag
           }

           // 4. Save to host temp
           tempDir, err := fs.TempDir("", "kinder-dev-load-")
           if err != nil { return fmt.Errorf("create temp dir: %w", err) }
           defer os.RemoveAll(tempDir)
           tarPath := filepath.Join(tempDir, "image.tar")
           if err := exec.CommandContext(ctx, opts.BinaryName, "save", "-o", tarPath, opts.ImageTag).Run(); err != nil {
               return fmt.Errorf("save image to tar: %w", err)
           }

           // 5. Concurrent per-node load
           fns := make([]func() error, 0, len(candidates))
           for _, n := range candidates {
               n := n
               fns = append(fns, func() error {
                   return loader(n, func() (io.ReadCloser, error) {
                       return os.Open(tarPath)
                   })
               })
           }
           return errors.UntilErrorConcurrent(fns)
       }

       // imageInspectID runs `<binary> image inspect -f {{ .Id }} <ref>`.
       // Wrapped as a package-level var for test injection.
       var imageInspectID = func(ctx context.Context, binaryName, ref string) (string, error) {
           lines, err := exec.OutputLines(exec.CommandContext(ctx, binaryName, "image", "inspect", "-f", "{{ .Id }}", ref))
           if err != nil { return "", err }
           if len(lines) == 0 { return "", fmt.Errorf("no image ID returned") }
           return lines[0], nil
       }
       ```

    1. **`rollout.go`**:
       ```go
       package dev

       import (
           "context"
           "fmt"
           "time"

           "sigs.k8s.io/kind/pkg/exec"
       )

       // RolloutRestartAndWait runs `kubectl rollout restart deployment/<name>`
       // followed by `kubectl rollout status --timeout=<dur> deployment/<name>`
       // against the host kubectl using the supplied kubeconfig file.
       //
       // This is the HOST-side rollout pattern (RESEARCH §3) — distinct from
       // the in-node pattern used by system addons during cluster create
       // (e.g. corednstuning.go uses node.CommandContext with the in-cluster
       // /etc/kubernetes/admin.conf). The user's Deployment lives in a user-
       // managed namespace, and `kinder dev` runs on the host, so we use the
       // host kubectl + the external (host-reachable) kubeconfig produced by
       // WriteKubeconfigTemp.
       func RolloutRestartAndWait(ctx context.Context, kubeconfigPath, namespace, deployment string, timeout time.Duration) error {
           if kubeconfigPath == "" { return fmt.Errorf("dev rollout: kubeconfigPath empty") }
           if namespace == ""      { return fmt.Errorf("dev rollout: namespace empty") }
           if deployment == ""     { return fmt.Errorf("dev rollout: deployment empty") }
           if timeout <= 0         { return fmt.Errorf("dev rollout: timeout %v must be > 0", timeout) }

           if err := exec.CommandContext(ctx,
               "kubectl",
               "--kubeconfig="+kubeconfigPath,
               "rollout", "restart",
               "--namespace="+namespace,
               "deployment/"+deployment,
           ).Run(); err != nil {
               return fmt.Errorf("rollout restart deployment/%s in namespace %s: %w", deployment, namespace, err)
           }

           if err := exec.CommandContext(ctx,
               "kubectl",
       ```
