       ```go
       package dev

       import (
           "context"
           "fmt"
           "time"
       )

       // runOneCycle executes one build → load → rollout cycle and prints
       // per-step timing to opts.streams.Out. On any step failure, returns
       // the wrapped error WITHOUT running subsequent steps. The cycle
       // header is printed BEFORE the build starts; the timing block is
       // printed after the cycle completes (on success or failure).
       //
       // Output format (RESEARCH §7):
       //   [cycle N] Change detected — starting build-load-rollout
       //     build:   X.Xs
       //     load:    X.Xs
       //     rollout: X.Xs
       //     total:   X.Xs
       //
       // On failure, the line for the failed step still prints with the
       // partial elapsed time, and a follow-up "(error: <msg>)" annotation
       // is appended. Subsequent step lines are NOT printed.
       func runOneCycle(ctx context.Context, opts cycleOpts) error {
           cycleStart := time.Now()
           fmt.Fprintf(opts.streams.Out, "[cycle %d] Change detected — starting build-load-rollout\n", opts.cycleNum)

           t0 := time.Now()
           if err := BuildImageFn(ctx, opts.binaryName, opts.imageTag, opts.watchDir); err != nil {
               fmt.Fprintf(opts.streams.Out, "  build:   %.1fs (error: %v)\n", time.Since(t0).Seconds(), err)
               return fmt.Errorf("build: %w", err)
           }
           buildDur := time.Since(t0)
           fmt.Fprintf(opts.streams.Out, "  build:   %.1fs\n", buildDur.Seconds())

           t1 := time.Now()
           if err := LoadImagesIntoCluster(ctx, LoadOptions{
               ClusterName: opts.clusterName,
               ImageTag:    opts.imageTag,
               BinaryName:  opts.binaryName,
               Logger:      opts.logger,
               Provider:    opts.provider,
           }); err != nil {
               fmt.Fprintf(opts.streams.Out, "  load:    %.1fs (error: %v)\n", time.Since(t1).Seconds(), err)
               return fmt.Errorf("load: %w", err)
           }
           loadDur := time.Since(t1)
           fmt.Fprintf(opts.streams.Out, "  load:    %.1fs\n", loadDur.Seconds())

           t2 := time.Now()
           if err := RolloutFn(ctx, opts.kubeconfigPath, opts.namespace, opts.target, opts.rolloutTimeout); err != nil {
               fmt.Fprintf(opts.streams.Out, "  rollout: %.1fs (error: %v)\n", time.Since(t2).Seconds(), err)
               return fmt.Errorf("rollout: %w", err)
           }
           rolloutDur := time.Since(t2)
           fmt.Fprintf(opts.streams.Out, "  rollout: %.1fs\n", rolloutDur.Seconds())

           fmt.Fprintf(opts.streams.Out, "  total:   %.1fs\n", time.Since(cycleStart).Seconds())
           return nil
       }
       ```

    1. **`dev.go`**:
       ```go
       package dev

       import (
           "context"
           "fmt"
           "os"
           "os/signal"
           "syscall"
           "time"

           "sigs.k8s.io/kind/pkg/cluster"
           "sigs.k8s.io/kind/pkg/cmd"
           "sigs.k8s.io/kind/pkg/log"
       )

       // Options carries the full configuration for one `kinder dev` invocation.
       // The CLI shell (Plan 04 / pkg/cmd/kind/dev/dev.go) builds this from
       // flags + cluster auto-detection and calls Run.
       type Options struct {
           // Optional ctx; nil = use context.Background wrapped with SIGINT/SIGTERM
           Ctx context.Context

           // Cluster targeting
           ClusterName string
           Provider    *cluster.Provider
           BinaryName  string // docker | podman | nerdctl

           // User inputs
           WatchDir  string
           Target    string // deployment name
           Namespace string
           ImageTag  string

           // Watch-mode tuning
           Debounce       time.Duration
           Poll           bool
           PollInterval   time.Duration
           RolloutTimeout time.Duration

           // I/O
           Logger  log.Logger
           Streams cmd.IOStreams

           // Test hooks
           EventSource      <-chan struct{} // nil = start real watcher/poller
           SkipBanner       bool
           ExitOnFirstError bool
       }

       // Run starts the watch-mode loop. Blocks until ctx is canceled (SIGINT,
       // SIGTERM, or caller's context). Returns nil on clean exit; returns the
       // first cycle error if ExitOnFirstError is set.
       func Run(opts Options) error {
           // 1. Validate required inputs
           if opts.WatchDir == ""    { return fmt.Errorf("kinder dev: --watch is required") }
           if opts.Target == ""      { return fmt.Errorf("kinder dev: --target is required") }
           if opts.ClusterName == "" { return fmt.Errorf("kinder dev: ClusterName is required (resolve via lifecycle.ResolveClusterName before calling Run)") }
           if opts.Provider == nil   { return fmt.Errorf("kinder dev: Provider is required") }
           if opts.BinaryName == ""  { return fmt.Errorf("kinder dev: BinaryName is required") }
           if opts.Namespace == ""      { opts.Namespace = "default" }
           if opts.Debounce == 0        { opts.Debounce = 500 * time.Millisecond }
           if opts.PollInterval == 0    { opts.PollInterval = time.Second }
           if opts.RolloutTimeout == 0  { opts.RolloutTimeout = 2 * time.Minute }
           if opts.ImageTag == ""       { opts.ImageTag = fmt.Sprintf("kinder-dev/%s:latest", opts.Target) }

           // 2. Wrap ctx with signal handling (RESEARCH common pitfall 6)
           parentCtx := opts.Ctx
           if parentCtx == nil { parentCtx = context.Background() }
           ctx, stop := signal.NotifyContext(parentCtx, os.Interrupt, syscall.SIGTERM)
           defer stop()

           // 3. Write kubeconfig ONCE at startup (RESEARCH anti-pattern: never per-cycle)
           kubeconfigPath, cleanup, err := WriteKubeconfigTemp(opts.Provider, opts.ClusterName)
           if err != nil { return fmt.Errorf("write kubeconfig tempfile: %w", err) }
           defer cleanup()

           // 4. Banner
           if !opts.SkipBanner {
               mode := "fsnotify"
               if opts.Poll { mode = "poll" }
               fmt.Fprintf(opts.Streams.Out, "Watching %s → deployment/%s (cluster: %s, namespace: %s)\n", opts.WatchDir, opts.Target, opts.ClusterName, opts.Namespace)
               fmt.Fprintf(opts.Streams.Out, "Debounce: %s | Mode: %s\n", opts.Debounce, mode)
               fmt.Fprintln(opts.Streams.Out, "Press Ctrl+C to exit.")
           }

           // 5. Start watcher (or use injected EventSource for tests)
           var rawEvents <-chan struct{}
           if opts.EventSource != nil {
               rawEvents = opts.EventSource
           } else if opts.Poll {
               rawEvents, err = StartPoller(ctx, opts.WatchDir, opts.PollInterval, opts.Logger)
           } else {
               rawEvents, err = StartWatcher(ctx, opts.WatchDir, opts.Logger)
           }
           if err != nil { return fmt.Errorf("start watcher: %w", err) }

           // 6. Pipe through debouncer
           cycles := Debounce(ctx, rawEvents, opts.Debounce)

           // 7. Drain cycles serially — at most one cycle in flight at a time
           cycleNum := 0
           var firstErr error
           for {
               select {
               case <-ctx.Done():
                   return firstErr
               case _, ok := <-cycles:
                   if !ok { return firstErr }
                   cycleNum++
                   cErr := runOneCycle(ctx, cycleOpts{
                       cycleNum:       cycleNum,
                       binaryName:     opts.BinaryName,
                       imageTag:       opts.ImageTag,
                       watchDir:       opts.WatchDir,
                       namespace:      opts.Namespace,
                       target:         opts.Target,
                       rolloutTimeout: opts.RolloutTimeout,
                       kubeconfigPath: kubeconfigPath,
                       clusterName:    opts.ClusterName,
                       provider:       opts.Provider,
                       logger:         opts.Logger,
                       streams:        opts.Streams,
                   })
                   if cErr != nil {
                       if opts.ExitOnFirstError {
                           return cErr
                       }
                       if firstErr == nil { firstErr = cErr }
                       fmt.Fprintf(opts.Streams.ErrOut, "cycle %d failed: %v\n", cycleNum, cErr)
                       fmt.Fprintln(opts.Streams.Out, "[watch] Ready. Watching for changes...")
                   } else {
                       fmt.Fprintln(opts.Streams.Out, "[watch] Ready. Watching for changes...")
                   }
                   // While runOneCycle was running, additional debounced events
                   // may have arrived. The cap=1 buffer of Debounce means at
                   // most one is queued. Drain it (non-blocking) so the user
                   // does not get an immediate redundant cycle right after
                   // a successful one — the debouncer was already debouncing
                   // throughout the cycle. RESEARCH common pitfall 3 says drop
                   // queued events while a cycle is in-flight.
                   select { case <-cycles: default: }
               }
           }
       }
       ```

    3. RED → GREEN. Run `go test -race ./pkg/internal/dev/... -run TestRun -count=1`.

    **Why kubeconfig is written ONCE not per-cycle:** RESEARCH anti-pattern. Per-cycle creation is N file ops per second of editing, and creates a leak surface if the per-cycle defer isn't pristine. Once-at-startup with a Run-level defer is the correct lifecycle.

    **Why we drain the cycles channel after each cycle:** During a long cycle (e.g. 30s build), the debouncer continues running and may have queued ONE event in its cap=1 output buffer. After the cycle completes, immediately running another cycle from that queued event is double-work — the user's edits already triggered the cycle that just completed. So we drain non-blockingly. Subsequent edits will fire fresh debounced events.

    **Why the loop's outer `select` includes `<-ctx.Done()`:** Even if the cycles channel is still open, ctx-cancel must terminate the loop immediately. Without this branch, the loop would block on `<-cycles` until the watcher's goroutine close cascade finishes — adds latency to Ctrl+C.
