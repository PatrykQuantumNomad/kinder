    func writeTruncatedTarGz(t *testing.T) string {
        t.Helper()
        var buf bytes.Buffer
        gw := gzip.NewWriter(&buf)
        tw := tar.NewWriter(gw)
        // Write header claiming 100 bytes
        hdr := &tar.Header{Name: "data.bin", Mode: 0o644, Size: 100}
        if err := tw.WriteHeader(hdr); err != nil {
            t.Fatalf("writing header: %v", err)

            t.Fatalf("writing partial body: %v", err)
        }
        // Force-close without completing — the tar is now truncated
        // Close gzip to flush, but the tar entry is incomplete
        gw.Close()
