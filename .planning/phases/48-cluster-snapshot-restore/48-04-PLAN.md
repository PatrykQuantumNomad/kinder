       // CONTEXT.md locked: K8s version, topology (CP count, worker count, LB

       func CheckCompatibility(snap, live *Metadata) error {
           var errs []error
           if snap.K8sVersion != live.K8sVersion {
               errs = append(errs, fmt.Errorf("%w: snapshot=%s cluster=%s",
                   ErrCompatK8sMismatch, snap.K8sVersion, live.K8sVersion))
           }
           if snap.Topology != live.Topology {
               errs = append(errs, fmt.Errorf("%w: snapshot=%+v cluster=%+v",
                   ErrCompatTopologyMismatch, snap.Topology, live.Topology))
           }
           if !addonsEqual(snap.AddonVersions, live.AddonVersions) {
               errs = append(errs, fmt.Errorf("%w: snapshot=%v cluster=%v (must match exactly — extras and version drifts both fail)",
                   ErrCompatAddonMismatch, snap.AddonVersions, live.AddonVersions))
           }
           return errors.NewAggregate(errs)
       }
       // addonsEqual: same key set AND identical values for every key. nil and
       // empty map treated as equal.

       ```go
       var ErrInsufficientDiskSpace = errors.New("insufficient disk space")

       // EnsureDiskSpace returns nil if the filesystem containing path has at
       // least required bytes free. Wraps ErrInsufficientDiskSpace with the
       // observed free bytes and the path's mount point.
       func EnsureDiskSpace(path string, required int64) error {
           var st syscall.Statfs_t
           if err := syscall.Statfs(path, &st); err != nil {
               return fmt.Errorf("statfs %s: %w", path, err)
           }
           // Bsize is the block size; Bavail is blocks available to non-root.
           free := int64(st.Bavail) * int64(st.Bsize)
           if free < required {
               return fmt.Errorf("%w: need %d bytes, %d available at %s",
                   ErrInsufficientDiskSpace, required, free, path)
           }
           return nil
       }
       ```
