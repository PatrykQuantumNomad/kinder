    ```go
    // DecodeOptions configures a RunDecode invocation.
    type DecodeOptions struct {
        Cluster             string        // resolved cluster name (display only)
        BinaryName          string        // "docker" / "podman" / "nerdctl"
        CPNodeName          string        // first control-plane node for kubectl exec
        AllNodes            []string      // all node container names (for docker logs scan)
        Since               time.Duration // locked decision #2: applied to both sources; default 30m enforced upstream by CLI
        IncludeNormalEvents bool          // locked decision #3 override; default false → Warnings only
        Logger              log.Logger    // optional; nil-safe via internal helper
    }

    // RunDecode collects docker logs from every node and kubectl events from
    // the CP node, matches both against Catalog, and returns the aggregated
    // DecodeResult. Failures from individual sources are logged and skipped;
    // the function only returns an error for caller misuse (e.g., empty inputs).
    func RunDecode(opts DecodeOptions) (*DecodeResult, error) {
        if len(opts.AllNodes) == 0 && opts.CPNodeName == "" {
            return nil, errors.New("RunDecode: AllNodes and CPNodeName both empty; nothing to scan")
        }
        sinceStr := opts.Since.String()  // "30m0s", "1h0m0s", "0s" if zero
        result := &DecodeResult{Cluster: opts.Cluster}
        var totalLines int

        // Per-node docker logs.
        for _, node := range opts.AllNodes {
            lines, err := dockerLogsFn(opts.BinaryName, node, sinceStr)
            if err != nil {
                if opts.Logger != nil {
                    opts.Logger.V(1).Infof("decode: skipping logs for %q: %v", node, err)
                }
                continue
            }
            totalLines += len(lines)
            result.Matches = append(result.Matches, matchLines(lines, Catalog, "docker-logs:"+node)...)
        }

        // kubectl events from CP.
        if opts.CPNodeName != "" {
            lines, err := k8sEventsFn(opts.BinaryName, opts.CPNodeName, sinceStr, opts.IncludeNormalEvents)
            if err != nil {
                if opts.Logger != nil {
                    opts.Logger.V(1).Infof("decode: skipping kubectl events: %v", err)
                }
            } else {
                totalLines += len(lines)
                result.Matches = append(result.Matches, matchLines(lines, Catalog, "k8s-events")...)
            }
        }
        result.Unmatched = totalLines - len(result.Matches)
        if result.Unmatched < 0 {
            result.Unmatched = 0 // defensive
        }
        return result, nil
    }
    ```

    Implementation notes:
    - Use `pkg/errors` (already in repo, used by lifecycle/state.go) for the empty-inputs error.
    - Use `sigs.k8s.io/kind/pkg/log` for the Logger interface (matches doctor.ApplySafeMitigations signature).
    - DO NOT thread context.Context through this signature — neither dockerLogsFn nor k8sEventsFn accept one (they shell out and return when the process exits). Adding ctx now would force fn-var sig change downstream; keep this surface minimal.
    - DO NOT shorten sinceStr to opts.Since.String()'s "30m" form by stripping "0s" — Docker accepts "30m0s" fine and tests need a deterministic exact-string comparison.
