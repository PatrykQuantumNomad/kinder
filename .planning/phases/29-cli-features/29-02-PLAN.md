   ```go
   func runE(logger log.Logger, streams cmd.IOStreams, flags *flagpole) error {
       // Validate output format
       switch flags.Output {
       case "", "json":
       default:
           return fmt.Errorf("unknown output format %q: supported values are \"\" and \"json\"", flags.Output)
       }

       provider := cluster.NewProvider(
           cluster.ProviderWithLogger(logger),
           runtime.GetDefault(logger),
       )

       var nodes []nodes.Node
       var err error
       if flags.AllClusters {
           clusters, err := provider.List()
           if err != nil {
               return err
           }
           for _, clusterName := range clusters {
               clusterNodes, err := provider.ListNodes(clusterName)
               if err != nil {
                   return err
               }
               nodes = append(nodes, clusterNodes...)
           }
       } else {
           nodes, err = provider.ListNodes(flags.Name)
           if err != nil {
               return err
           }
       }

       // JSON output
       if flags.Output == "json" {
           infos := make([]nodeInfo, 0, len(nodes))
           for _, n := range nodes {
               role, err := n.Role()
               if err != nil {
                   role = "unknown"
               }
               infos = append(infos, nodeInfo{Name: n.String(), Role: role})
           }
           return json.NewEncoder(streams.Out).Encode(infos)
       }

       // Human-readable output
       if len(nodes) == 0 {
           if flags.AllClusters {
               logger.V(0).Infof("No kind nodes for any cluster.")
           } else {
               logger.V(0).Infof("No kind nodes found for cluster %q.", flags.Name)
           }
           return nil
       }
       for _, node := range nodes {
           fmt.Fprintln(streams.Out, node.String()) //nolint:errcheck
       }
       return nil
   }
   ```

2. Add the `CreateWithAddonProfile` function at the end of the file:
   ```go
   // CreateWithAddonProfile applies a named addon preset to the cluster configuration.
   // If profile is "" (empty), this is a no-op and all addons remain enabled by default.
   // Profile overrides any addon settings from --config when both are specified.
   // Valid profiles: "minimal" (no addons), "full" (all addons), "gateway" (MetalLB + EnvoyGateway), "ci" (MetricsServer + CertManager).
   func CreateWithAddonProfile(profile string) CreateOption {
       return createOptionAdapter(func(o *internalcreate.ClusterOptions) error {
           if profile == "" {
               return nil
           }
           // Ensure Config is initialized — options run before fixupOptions,
           // so o.Config may be nil when no --config flag is passed.
           if o.Config == nil {
               cfg, err := internalencoding.Load("")
               if err != nil {
                   return err
               }
               o.Config = cfg
           }
           switch profile {
           case "minimal":
               o.Config.Addons = internalconfig.Addons{} // zero value = all false
           case "full":
               o.Config.Addons = internalconfig.Addons{
                   MetalLB:       true,
                   EnvoyGateway:  true,
                   MetricsServer: true,
                   CoreDNSTuning: true,
                   Dashboard:     true,
                   LocalRegistry: true,
                   CertManager:   true,
               }
           case "gateway":
               o.Config.Addons = internalconfig.Addons{
                   MetalLB:      true,
                   EnvoyGateway: true,
               }
           case "ci":
               o.Config.Addons = internalconfig.Addons{
                   MetricsServer: true,
                   CertManager:   true,
               }
           default:
               return fmt.Errorf("unknown profile %q: valid values are minimal, full, gateway, ci", profile)
           }
           return nil
       })
   }
   ```
